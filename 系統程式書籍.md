## 第一章：系統程式概述

### 1.1 什麼是系統程式

系統程式（System Programming）是指直接與電腦系統硬體和核心軟體互動的程式設計領域。與應用程式開發不同，系統程式設計需要深入了解作業系統的內部機制、硬體架構以及低階記憶體管理。

系統程式設計的特點包括：

- **低階存取**：直接操作記憶體、暫存器和硬體
- **效能導向**：追求最高的執行效率和最小的資源消耗
- **硬體相依性**：需要了解特定平台的硬體特性
- **穩定性要求**：系統程式通常需要長時間穩定運作

### 1.2 系統程式的應用領域

系統程式廣泛應用於以下領域：

- **作業系統核心**：Windows、Linux、macOS 的核心元件
- **驅動程式**：硬體裝置的軟體介面
- **編譯器與直譯器**：程式語言的翻譯工具
- **資料庫系統**：高效能的資料儲存與檢索
- **網路服務**：伺服器、負載平衡器、防火牆
- **嵌入式系統**：物聯網裝置、韌體開發

### 1.3 學習路徑

本書籍將帶您從基礎概念出發，逐步深入系統程式設計的核心。我們將使用 C 語言作為主要語言，因為它是系統程式設計的經典選擇，能夠提供對硬體的直接控制能力，同時保持足夠的可攜性。

### 1.4 開發環境設置

在開始之前，讓我們建立一個適合系統程式開發的環境。以下範例適用於 Linux 系統：

```c
#include <stdio.h>
#include <stdlib.h>

int main() {
    printf("系統程式開發環境測試\n");
    printf("編譯器：%s\n", __VERSION__);
    printf("平台：%s\n", 
#if defined(__linux__)
        "Linux"
#elif defined(_WIN32)
        "Windows"
#elif defined(__APPLE__)
        "macOS"
#endif
    );
    return 0;
}
```

編譯方式：

```bash
gcc -Wall -Wextra -o test test.c
./test
```

建議的編譯選項：

```bash
gcc -Wall -Wextra -Werror -std=c11 -O2 -g -o program program.c
```

- `-Wall -Wextra`：開啟額外警告
- `-Werror`：將警告視為錯誤
- `-std=c11`：使用 C11 標準
- `-O2`：最佳化等級
- `-g`：加入除錯資訊

## 第二章：C 語言核心概念

### 2.1 指標與位址

指標是 C 語言中最強大也最危險的特性。在系統程式設計中，我們經常需要直接操作記憶體位址。

```c
#include <stdio.h>
#include <stdlib.h>

int main() {
    int value = 42;
    int *ptr = &value;
    
    printf("value 的值：%d\n", value);
    printf("value 的位址：%p\n", (void*)&value);
    printf("ptr 儲存的值：%p\n", (void*)ptr);
    printf("透過 ptr 存取的值：%d\n", *ptr);
    
    return 0;
}
```

**指標算術**：指標運算會根據指標類型自動調整位移。

```c
#include <stdio.h>

int main() {
    int arr[] = {10, 20, 30, 40, 50};
    int *p = arr;
    
    for (int i = 0; i < 5; i++) {
        printf("arr[%d] = %d, 位址 = %p\n", i, *(p + i), (void*)(p + i));
    }
    
    printf("\n指標遞增演示：\n");
    p = arr;
    while (p < arr + 5) {
        printf("%d ", *p++);
    }
    printf("\n");
    
    return 0;
}
```

### 2.2 指標的類型

**void 指標**：通用指標類型，可以指向任何類型的資料。

```c
#include <stdio.h>

void print_bytes(void *ptr, size_t size) {
    unsigned char *byte = (unsigned char *)ptr;
    printf("記憶體內容：");
    for (size_t i = 0; i < size; i++) {
        printf("%02x ", byte[i]);
    }
    printf("\n");
}

int main() {
    int value = 0x12345678;
    print_bytes(&value, sizeof(int));
    
    double d = 3.14159;
    print_bytes(&d, sizeof(double));
    
    return 0;
}
```

**函數指標**：儲存函數位址的指標。

```c
#include <stdio.h>

int add(int a, int b) { return a + b; }
int subtract(int a, int b) { return a - b; }
int multiply(int a, int b) { return a * b; }

int main() {
    int (*operation)(int, int);
    
    operation = add;
    printf("10 + 5 = %d\n", operation(10, 5));
    
    operation = subtract;
    printf("10 - 5 = %d\n", operation(10, 5));
    
    operation = multiply;
    printf("10 * 5 = %d\n", operation(10, 5));
    
    return 0;
}
```

### 2.3 結構體與記憶體對齊

結構體是系統程式設計中常用的資料結構，用於封裝相關的異質資料。

```c
#include <stdio.h>
#include <stddef.h>

struct example {
    char a;      // 1 位元組
    int b;       // 4 位元組（通常從 4 位元組邊界開始）
    char c;      // 1 位元組
    double d;    // 8 位元組（通常從 8 位元組邊界開始）
};

int main() {
    struct example ex = {'A', 100, 'B', 3.14};
    
    printf("結構體大小：%zu 位元組\n", sizeof(struct example));
    printf("成員偏移量：\n");
    printf("  a: %zu\n", offsetof(struct example, a));
    printf("  b: %zu\n", offsetof(struct example, b));
    printf("  c: %zu\n", offsetof(struct example, c));
    printf("  d: %zu\n", offsetof(struct example, d));
    
    return 0;
}
```

**記憶體對齊的考量**：


### 2.4 位元操作

位元操作是系統程式設計中的核心技能，用於低位階的資料處理。

```c
#include <stdio.h>

void print_binary(unsigned int value) {
    for (int i = 31; i >= 0; i--) {
        printf("%d", (value >> i) & 1);
        if (i % 8 == 0) printf(" ");
    }
    printf("\n");
}

int main() {
    unsigned int a = 0b11001010;
    unsigned int b = 0b10110110;
    
    printf("a = "); print_binary(a);
    printf("b = "); print_binary(b);
    
    printf("\na & b = "); print_binary(a & b);  // AND
    printf("a | b = "); print_binary(a | b);  // OR
    printf("a ^ b = "); print_binary(a ^ b);  // XOR
    printf("~a    = "); print_binary(~a);      // NOT
    
    printf("\na << 2 = "); print_binary(a << 2);
    printf("a >> 3 = "); print_binary(a >> 3);
    
    return 0;
}
```

**常見位元操作技巧**：

```c
#include <stdio.h>
#include <stdint.h>

int main() {
    unsigned int flags = 0;
    
    // 設定特定位元
    flags |= (1 << 3);  // 設定第 3 位元
    printf("設定後：%08x\n", flags);
    
    // 清除特定位元
    flags &= ~(1 << 3);  // 清除第 3 位元
    printf("清除後：%08x\n", flags);
    
    // 切換特定位元
    flags ^= (1 << 3);  // 切換第 3 位元
    printf("切換後：%08x\n", flags);
    
    // 檢查特定位元
    if (flags & (1 << 3)) {
        printf("第 3 位元已設定\n");
    }
    
    // 取得低位連續位元
    unsigned int value = 0b00010100;
    printf("低位連續位元：%u\n", value & -value);
    
    // 計算有多少位元被設定
    unsigned int count = __builtin_popcount(value);
    printf("設定的位元數：%u\n", count);
    
    return 0;
}
```

## 第三章：記憶體管理

### 3.1 記憶體配置

C 語言提供了多種記憶體配置方式，每種方式適用於不同的場景。

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main() {
    // 靜態配置
    int static_array[100];
    printf("靜態陣列大小：%zu 位元組\n", sizeof(static_array));
    
    // 動態配置 - malloc
    int *dynamic = (int *)malloc(100 * sizeof(int));
    if (dynamic == NULL) {
        fprintf(stderr, "記憶體配置失敗\n");
        return 1;
    }
    
    // 使用 calloc 配置並初始化為零
    int *calloc_array = (int *)calloc(100, sizeof(int));
    if (calloc_array == NULL) {
        free(dynamic);
        fprintf(stderr, "記憶體配置失敗\n");
        return 1;
    }
    
    // 配置並初始化
    char *str = (char *)malloc(50);
    strcpy(str, "Hello, World!");
    printf("字串內容：%s\n", str);
    
    // 重新配置 - realloc
    int *reallocated = (int *)realloc(dynamic, 200 * sizeof(int));
    if (reallocated == NULL) {
        free(dynamic);
        free(calloc_array);
        free(str);
        fprintf(stderr, "記憶體重新配置失敗\n");
        return 1;
    }
    dynamic = reallocated;
    
    // 釋放記憶體
    free(dynamic);
    free(calloc_array);
    free(str);
    
    printf("記憶體配置與釋放完成\n");
    return 0;
}
```

### 3.2 記憶體池配置

在系統程式設計中，自定義記憶體池可以提高效能並減少記憶體碎片。

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define POOL_SIZE 4096
#define BLOCK_SIZE 64

typedef struct block_header {
    size_t size;
    int free;
    struct block_header *next;
} block_header_t;

typedef struct memory_pool {
    void *memory;
    block_header_t *first_block;
    size_t total_size;
} memory_pool_t;

memory_pool_t *pool_create(size_t size) {
    memory_pool_t *pool = (memory_pool_t *)malloc(sizeof(memory_pool_t));
    if (!pool) return NULL;
    
    pool->memory = malloc(size);
    if (!pool->memory) {
        free(pool);
        return NULL;
    }
    
    pool->total_size = size;
    pool->first_block = (block_header_t *)pool->memory;
    pool->first_block->size = size - sizeof(block_header_t);
    pool->first_block->free = 1;
    pool->first_block->next = NULL;
    
    return pool;
}

void *pool_alloc(memory_pool_t *pool, size_t size) {
    block_header_t *current = pool->first_block;
    
    while (current) {
        if (current->free && current->size >= size) {
            size_t remaining = current->size - size;
            
            if (remaining > sizeof(block_header_t)) {
                block_header_t *new_block = (block_header_t *)((char *)current + sizeof(block_header_t) + size);
                new_block->size = remaining - sizeof(block_header_t);
                new_block->free = 1;
                new_block->next = current->next;
                
                current->size = size;
                current->next = new_block;
            }
            
            current->free = 0;
            return (void *)((char *)current + sizeof(block_header_t));
        }
        current = current->next;
    }
    
    return NULL;
}

void pool_free(memory_pool_t *pool, void *ptr) {
    if (!ptr) return;
    
    block_header_t *block = (block_header_t *)((char *)ptr - sizeof(block_header_t));
    block->free = 1;
    
    // 合併相鄰的可用區塊
    block_header_t *current = pool->first_block;
    while (current && current->next) {
        if (current->free && current->next->free) {
            current->size += sizeof(block_header_t) + current->next->size;
            current->next = current->next->next;
        }
        current = current->next;
    }
}

void pool_destroy(memory_pool_t *pool) {
    free(pool->memory);
    free(pool);
}

int main() {
    memory_pool_t *pool = pool_create(POOL_SIZE);
    if (!pool) {
        fprintf(stderr, "記憶體池建立失敗\n");
        return 1;
    }
    
    printf("記憶體池大小：%zu 位元組\n", pool->total_size);
    
    void *p1 = pool_alloc(pool, 100);
    void *p2 = pool_alloc(pool, 200);
    void *p3 = pool_alloc(pool, 150);
    
    printf("配置 p1：%p\n", p1);
    printf("配置 p2：%p\n", p2);
    printf("配置 p3：%p\n", p3);
    
    pool_free(pool, p2);
    pool_free(pool, p3);
    
    void *p4 = pool_alloc(pool, 300);
    printf("再次配置 p4：%p\n", p4);
    
    pool_destroy(pool);
    return 0;
}
```

### 3.3 記憶體映射

記憶體映射允許程式直接存取檔案作為記憶體，這在系統程式設計中非常有用。

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <sys/mman.h>
#include <sys/stat.h>
#include <fcntl.h>
#include <unistd.h>

int main() {
    const char *filename = "mmap_test.txt";
    const char *content = "這是記憶體映射檔案的測試內容。\n";
    size_t size = strlen(content);
    
    // 建立測試檔案
    int fd = open(filename, O_RDWR | O_CREAT | O_TRUNC, 0644);
    if (fd == -1) {
        perror("open");
        return 1;
    }
    
    // 擴展檔案大小
    if (ftruncate(fd, size) == -1) {
        perror("ftruncate");
        close(fd);
        return 1;
    }
    
    // 建立記憶體映射
    void *mapped = mmap(NULL, size, PROT_READ | PROT_WRITE, MAP_SHARED, fd, 0);
    if (mapped == MAP_FAILED) {
        perror("mmap");
        close(fd);
        return 1;
    }
    
    // 關閉檔案描述符（映射仍然有效）
    close(fd);
    
    // 寫入資料（直接寫入記憶體）
    memcpy(mapped, content, size);
    
    // 同步記憶體與檔案
    if (msync(mapped, size, MS_SYNC) == -1) {
        perror("msync");
    }
    
    // 驗證寫入
    printf("映射記憶體內容：%s", (char *)mapped);
    
    // 解除映射
    if (munmap(mapped, size) == -1) {
        perror("munmap");
        return 1;
    }
    
    // 重新開啟驗證
    fd = open(filename, O_RDONLY);
    char buffer[256];
    ssize_t n = read(fd, buffer, sizeof(buffer) - 1);
    buffer[n] = '\0';
    printf("檔案內容驗證：%s", buffer);
    close(fd);
    
    unlink(filename);
    return 0;
}
```

### 3.5 常見記憶體錯誤與防範

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// 安全的記憶體配置包裝函數
void *safe_malloc(size_t size, const char *context) {
    void *ptr = malloc(size);
    if (ptr == NULL) {
        fprintf(stderr, "記憶體配置失敗：%s (需要 %zu 位元組)\n", context, size);
        exit(EXIT_FAILURE);
    }
    return ptr;
}

void *safe_calloc(size_t nmemb, size_t size, const char *context) {
    void *ptr = calloc(nmemb, size);
    if (ptr == NULL) {
        fprintf(stderr, "記憶體配置失敗：%s (需要 %zu 個元素，每個 %zu 位元組)\n", 
                context, nmemb, size);
        exit(EXIT_FAILURE);
    }
    return ptr;
}

void *safe_realloc(void *ptr, size_t size, const char *context) {
    void *new_ptr = realloc(ptr, size);
    if (new_ptr == NULL) {
        fprintf(stderr, "記憶體重新配置失敗：%s\n", context);
        free(ptr);
        exit(EXIT_FAILURE);
    }
    return new_ptr;
}

int main() {
    // 使用安全的配置函數
    int *numbers = (int *)safe_calloc(10, sizeof(int), "numbers 陣列");
    
    for (int i = 0; i < 10; i++) {
        numbers[i] = i * 10;
        printf("numbers[%d] = %d\n", i, numbers[i]);
    }
    
    // 安全的重新配置
    numbers = (int *)safe_realloc(numbers, 20 * sizeof(int), "numbers 擴展");
    
    for (int i = 10; i < 20; i++) {
        numbers[i] = i * 10;
    }
    
    printf("\n擴展後：numbers[15] = %d\n", numbers[15]);
    
    // 安全的字串複製
    const char *source = "這是一個測試字串";
    size_t len = strlen(source) + 1;
    char *dest = (char *)safe_malloc(len, "dest 字串");
    strncpy(dest, source, len);
    dest[len - 1] = '\0';  // 確保 null 終止
    
    printf("複製結果：%s\n", dest);
    
    free(numbers);
    free(dest);
    
    return 0;
}
```

## 第四章：行程與執行序

### 4.1 行程基礎

行程是作業系統進行資源分配和排程的基本單位。每個行程都有自己獨立的位址空間、檔案描述符和其他系統資源。

```c
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <sys/types.h>
#include <sys/wait.h>

int main() {
    pid_t pid = fork();
    
    if (pid < 0) {
        perror("fork 失敗");
        exit(EXIT_FAILURE);
    }
    
    if (pid == 0) {
        // 子行程
        printf("子行程：PID = %d, 父行程 PID = %d\n", getpid(), getppid());
        
        // 執行新程式
        execlp("ls", "ls", "-la", NULL);
        
        // 如果 execlp 失敗
        perror("execlp 失敗");
        exit(EXIT_FAILURE);
    } else {
        // 父行程
        int status;
        waitpid(pid, &status, 0);
        
        printf("父行程：PID = %d, 子行程 PID = %d\n", getpid(), pid);
        
        if (WIFEXITED(status)) {
            printf("子行程正常結束，返回值：%d\n", WEXITSTATUS(status));
        }
    }
    
    return 0;
}
```

### 4.2 行程的創建與管理

```c
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <sys/types.h>
#include <sys/wait.h>
#include <string.h>

#define MAX_PROCESSES 4

int main() {
    pid_t processes[MAX_PROCESSES];
    
    for (int i = 0; i < MAX_PROCESSES; i++) {
        pid_t pid = fork();
        
        if (pid < 0) {
            perror("fork 失敗");
            for (int j = 0; j < i; j++) {
                kill(processes[j], SIGTERM);
            }
            exit(EXIT_FAILURE);
        }
        
        if (pid == 0) {
            // 子行程執行工作
            printf("子行程 %d 開始工作 (PID: %d)\n", i, getpid());
            sleep(1);
            printf("子行程 %d 完成工作\n", i);
            exit(i);  // 返回索引作為退出碼
        }
        
        processes[i] = pid;
        printf("父行程：創建了子行程 %d，PID = %d\n", i, pid);
    }
    
    // 等待所有子行程
    for (int i = 0; i < MAX_PROCESSES; i++) {
        int status;
        pid_t pid = waitpid(processes[i], &status, 0);
        
        if (WIFEXITED(status)) {
            printf("子行程 %d (PID: %d) 已結束，返回值：%d\n", 
                   i, pid, WEXITSTATUS(status));
        }
    }
    
    printf("所有子行程已完成\n");
    return 0;
}
```

### 4.3 執行緒基礎

執行緒是行程內的執行單位，與行程相比，執行緒共享大部分的資源，創建和切換成本較低。

```c
#include <stdio.h>
#include <stdlib.h>
#include <pthread.h>
#include <unistd.h>

void *thread_function(void *arg) {
    int id = *(int *)arg;
    printf("執行緒 %d 開始執行 (TID: %lu)\n", id, (unsigned long)pthread_self());
    
    for (int i = 0; i < 5; i++) {
        printf("執行緒 %d 迭代 %d\n", id, i + 1);
        sleep(1);
    }
    
    printf("執行緒 %d 完成\n", id);
    
    int *result = malloc(sizeof(int));
    *result = id * 100;
    pthread_exit(result);
}

int main() {
    pthread_t threads[3];
    int thread_ids[3] = {1, 2, 3};
    
    printf("主執行緒開始 (TID: %lu)\n", (unsigned long)pthread_self());
    
    // 創建執行緒
    for (int i = 0; i < 3; i++) {
        if (pthread_create(&threads[i], NULL, thread_function, &thread_ids[i]) != 0) {
            perror("pthread_create 失敗");
            exit(EXIT_FAILURE);
        }
        printf("創建了執行緒 %d，TID = %lu\n", i + 1, (unsigned long)threads[i]);
    }
    
    // 等待執行緒結束並取得返回值
    for (int i = 0; i < 3; i++) {
        void *result;
        pthread_join(threads[i], &result);
        printf("執行緒 %d 已加入，返回值：%d\n", i + 1, *(int *)result);
        free(result);
    }
    
    printf("所有執行緒已完成\n");
    return 0;
}
```

### 4.4 執行緒同步

執行緒同步是防止資料競爭和確保執行緒安全的關鍵機制。

```c
#include <stdio.h>
#include <stdlib.h>
#include <pthread.h>
#include <unistd.h>

#define NUM_THREADS 5
#define NUM_ITERATIONS 10

// 共享資源
int shared_counter = 0;

// 互斥鎖
pthread_mutex_t mutex = PTHREAD_MUTEX_INITIALIZER;

// 讀寫鎖
pthread_rwlock_t rwlock = PTHREAD_RWLOCK_INITIALIZER;
int shared_data = 0;

// 使用互斥鎖保護共享資源
void *mutex_thread(void *arg) {
    int id = *(int *)arg;
    
    for (int i = 0; i < NUM_ITERATIONS; i++) {
        pthread_mutex_lock(&mutex);
        shared_counter++;
        printf("執行緒 %d：counter = %d\n", id, shared_counter);
        pthread_mutex_unlock(&mutex);
        usleep(10000);  // 10ms
    }
    
    return NULL;
}

// 使用讀寫鎖
void *reader_thread(void *arg) {
    int id = *(int *)arg;
    
    for (int i = 0; i < NUM_ITERATIONS; i++) {
        pthread_rwlock_rdlock(&rwlock);
        printf("讀者 %d：data = %d\n", id, shared_data);
        pthread_rwlock_unlock(&rwlock);
        usleep(10000);
    }
    
    return NULL;
}

void *writer_thread(void *arg) {
    int id = *(int *)arg;
    
    for (int i = 0; i < NUM_ITERATIONS; i++) {
        pthread_rwlock_wrlock(&rwlock);
        shared_data++;
        printf("寫者 %d：data = %d\n", id, shared_data);
        pthread_rwlock_unlock(&rwlock);
        usleep(20000);  // 寫入需要更長時間
    }
    
    return NULL;
}

int main() {
    pthread_t threads[NUM_THREADS];
    int ids[NUM_THREADS];
    
    printf("=== 互斥鎖範例 ===\n");
    shared_counter = 0;
    
    for (int i = 0; i < NUM_THREADS; i++) {
        ids[i] = i + 1;
        pthread_create(&threads[i], NULL, mutex_thread, &ids[i]);
    }
    
    for (int i = 0; i < NUM_THREADS; i++) {
        pthread_join(threads[i], NULL);
    }
    
    printf("最終計數器值：%d (預期：%d)\n", shared_counter, NUM_THREADS * NUM_ITERATIONS);
    
    // 讀寫鎖範例
    printf("\n=== 讀寫鎖範例 ===\n");
    shared_data = 0;
    
    for (int i = 0; i < 3; i++) {
        ids[i] = i + 1;
        pthread_create(&threads[i], NULL, reader_thread, &ids[i]);
    }
    
    for (int i = 3; i < 5; i++) {
        ids[i] = i + 1;
        pthread_create(&threads[i], NULL, writer_thread, &ids[i]);
    }
    
    for (int i = 0; i < NUM_THREADS; i++) {
        pthread_join(threads[i], NULL);
    }
    
    pthread_mutex_destroy(&mutex);
    pthread_rwlock_destroy(&rwlock);
    
    return 0;
}
```

### 4.5 條件變數與生產者-消費者問題

```c
#include <stdio.h>
#include <stdlib.h>
#include <pthread.h>
#include <unistd.h>
#include <semaphore.h>

#define BUFFER_SIZE 5

typedef struct {
    int buffer[BUFFER_SIZE];
    int count;
    int in;
    int out;
    pthread_mutex_t mutex;
    pthread_cond_t not_empty;
    pthread_cond_t not_full;
} shared_buffer_t;

void buffer_init(shared_buffer_t *sb) {
    sb->count = 0;
    sb->in = 0;
    sb->out = 0;
    pthread_mutex_init(&sb->mutex, NULL);
    pthread_cond_init(&sb->not_empty, NULL);
    pthread_cond_init(&sb->not_full, NULL);
}

void buffer_destroy(shared_buffer_t *sb) {
    pthread_mutex_destroy(&sb->mutex);
    pthread_cond_destroy(&sb->not_empty);
    pthread_cond_destroy(&sb->not_full);
}

void produce(shared_buffer_t *sb, int item) {
    pthread_mutex_lock(&sb->mutex);
    
    while (sb->count >= BUFFER_SIZE) {
        pthread_cond_wait(&sb->not_full, &sb->mutex);
    }
    
    sb->buffer[sb->in] = item;
    sb->in = (sb->in + 1) % BUFFER_SIZE;
    sb->count++;
    
    printf("生產者：放入 %d，緩衝區數量：%d\n", item, sb->count);
    
    pthread_cond_signal(&sb->not_empty);
    pthread_mutex_unlock(&sb->mutex);
}

int consume(shared_buffer_t *sb) {
    pthread_mutex_lock(&sb->mutex);
    
    while (sb->count <= 0) {
        pthread_cond_wait(&sb->not_empty, &sb->mutex);
    }
    
    int item = sb->buffer[sb->out];
    sb->out = (sb->out + 1) % BUFFER_SIZE;
    sb->count--;
    
    printf("消費者：取出 %d，緩衝區數量：%d\n", item, sb->count);
    
    pthread_cond_signal(&sb->not_full);
    pthread_mutex_unlock(&sb->mutex);
    
    return item;
}

void *producer_thread(void *arg) {
    shared_buffer_t *sb = (shared_buffer_t *)arg;
    
    for (int i = 1; i <= 10; i++) {
        produce(sb, i * 100);
        usleep(50000);  // 50ms
    }
    
    return NULL;
}

void *consumer_thread(void *arg) {
    shared_buffer_t *sb = (shared_buffer_t *)arg;
    
    for (int i = 0; i < 10; i++) {
        int item = consume(sb);
        usleep(80000);  // 80ms
    }
    
    return NULL;
}

int main() {
    shared_buffer_t shared;
    buffer_init(&shared);
    
    pthread_t prod, cons;
    
    pthread_create(&prod, NULL, producer_thread, &shared);
    pthread_create(&cons, NULL, consumer_thread, &shared);
    
    pthread_join(prod, NULL);
    pthread_join(cons, NULL);
    
    buffer_destroy(&shared);
    
    printf("生產者-消費者問題完成\n");
    return 0;
}
```

### 5.1 管道

管道是最基本的行程間通訊機制，提供單向的位元組流。

```c
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <string.h>
#include <sys/wait.h>

int main() {
    int pipefd[2];
    
    if (pipe(pipefd) == -1) {
        perror("pipe");
        exit(EXIT_FAILURE);
    }
    
    pid_t pid = fork();
    
    if (pid == -1) {
        perror("fork");
        exit(EXIT_FAILURE);
    }
    
    if (pid == 0) {
        // 子行程：關閉讀端
        close(pipefd[0]);
        
        // 寫入資料
        const char *message = "來自子行程的訊息！\n";
        write(pipefd[1], message, strlen(message));
        
        close(pipefd[1]);
        exit(EXIT_SUCCESS);
    } else {
        // 父行程：關閉寫端
        close(pipefd[1]);
        
        // 讀取資料
        char buffer[256];
        ssize_t n = read(pipefd[0], buffer, sizeof(buffer) - 1);
        
        if (n > 0) {
            buffer[n] = '\0';
            printf("父行程收到：%s", buffer);
        }
        
        close(pipefd[0]);
        
        int status;
        waitpid(pid, &status, 0);
    }
    
    return 0;
}
```

### 5.2 命名管道（FIFO）

命名管道允許無相關的行程之間進行通訊。

```c
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <sys/stat.h>
#include <sys/wait.h>
#include <string.h>
#include <fcntl.h>

#define FIFO_PATH "/tmp/my_fifo"

int main() {
    // 建立 FIFO
    mkfifo(FIFO_PATH, 0666);
    
    pid_t pid = fork();
    
    if (pid == 0) {
        // 子行程：作為寫入者
        int fd = open(FIFO_PATH, O_WRONLY);
        
        if (fd == -1) {
            perror("open");
            exit(EXIT_FAILURE);
        }
        
        for (int i = 1; i <= 5; i++) {
            char buffer[128];
            snprintf(buffer, sizeof(buffer), "訊息 %d 從子行程發送\n", i);
            write(fd, buffer, strlen(buffer));
            printf("子行程：已發送訊息 %d\n", i);
            sleep(1);
        }
        
        close(fd);
        exit(EXIT_SUCCESS);
    } else {
        // 父行程：作為讀取者
        int fd = open(FIFO_PATH, O_RDONLY);
        
        if (fd == -1) {
            perror("open");
            unlink(FIFO_PATH);
            exit(EXIT_FAILURE);
        }
        
        char buffer[128];
        ssize_t n;
        
        while ((n = read(fd, buffer, sizeof(buffer) - 1)) > 0) {
            buffer[n] = '\0';
            printf("父行程收到：%s", buffer);
        }
        
        close(fd);
        
        waitpid(pid, NULL, 0);
        unlink(FIFO_PATH);
    }
    
    return 0;
}
```

### 5.3 共享記憶體

共享記憶體是最高效的行程間通訊方式，允許多個行程直接存取同一塊記憶體。

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <sys/shm.h>
#include <sys/ipc.h>
#include <sys/wait.h>
#include <unistd.h>

#define SHM_KEY 0x1234
#define SHM_SIZE 1024

typedef struct {
    int data;
    pthread_mutex_t mutex;
} shared_data_t;

int main() {
    // 建立共享記憶體
    int shmid = shmget(SHM_KEY, SHM_SIZE, IPC_CREAT | 0666);
    
    if (shmid == -1) {
        perror("shmget");
        exit(EXIT_FAILURE);
    }
    
    // 附加共享記憶體
    shared_data_t *shm = (shared_data_t *)shmat(shmid, NULL, 0);
    
    if (shm == (void *)-1) {
        perror("shmat");
        shmctl(shmid, IPC_RMID, NULL);
        exit(EXIT_FAILURE);
    }
    
    // 初始化
    shm->data = 0;
    pthread_mutexattr_t attr;
    pthread_mutexattr_init(&attr);
    pthread_mutexattr_setpshared(&attr, PTHREAD_PROCESS_SHARED);
    pthread_mutex_init(&shm->mutex, &attr);
    
    pid_t pid = fork();
    
    if (pid == 0) {
        // 子行程：寫入共享記憶體
        for (int i = 0; i < 10; i++) {
            pthread_mutex_lock(&shm->mutex);
            shm->data++;
            printf("子行程：data = %d\n", shm->data);
            pthread_mutex_unlock(&shm->mutex);
            usleep(50000);
        }
        
        shmdt(shm);
        exit(EXIT_SUCCESS);
    } else {
        // 父行程：讀取共享記憶體
        for (int i = 0; i < 10; i++) {
            usleep(60000);
            pthread_mutex_lock(&shm->mutex);
            printf("父行程：data = %d\n", shm->data);
            pthread_mutex_unlock(&shm->mutex);
        }
        
        waitpid(pid, NULL, 0);
        
        printf("最終資料值：%d\n", shm->data);
        
        // 分離並刪除共享記憶體
        shmdt(shm);
        shmctl(shmid, IPC_RMID, NULL);
    }
    
    return 0;
}
```

### 5.4 訊號量

訊號量用於控制對共享資源的訪問，是行程同步的重要機制。

```c
#include <stdio.h>
#include <stdlib.h>
#include <sys/sem.h>
#include <sys/wait.h>
#include <unistd.h>
#include <string.h>

union semun {
    int val;
    struct semid_ds *buf;
    unsigned short *array;
};

// 訊號量操作
void sem_op(int semid, int op) {
    struct sembuf sb;
    sb.sem_num = 0;
    sb.sem_op = op;
    sb.sem_flg = 0;
    
    if (semop(semid, &sb, 1) == -1) {
        perror("semop");
        exit(EXIT_FAILURE);
    }
}

int main() {
    // 建立訊號量
    int semid = semget(IPC_PRIVATE, 1, 0666);
    
    if (semid == -1) {
        perror("semget");
        exit(EXIT_FAILURE);
    }
    
    // 初始化訊號量為 1
    union semun arg;
    arg.val = 1;
    
    if (semctl(semid, 0, SETVAL, arg) == -1) {
        perror("semctl");
        exit(EXIT_FAILURE);
    }
    
    pid_t pid = fork();
    
    if (pid == 0) {
        // 子行程
        printf("子行程：等待訊號量...\n");
        sem_op(semid, -1);  // P 操作
        
        printf("子行程：進入臨界區\n");
        for (int i = 0; i < 3; i++) {
            printf("子行程：工作中 %d\n", i + 1);
            sleep(1);
        }
        printf("子行程：離開臨界區\n");
        
        sem_op(semid, 1);  // V 操作
        
        exit(EXIT_SUCCESS);
    } else {
        // 父行程
        printf("父行程：等待訊號量...\n");
        sem_op(semid, -1);  // P 操作
        
        printf("父行程：進入臨界區\n");
        for (int i = 0; i < 3; i++) {
            printf("父行程：工作中 %d\n", i + 1);
            sleep(1);
        }
        printf("父行程：離開臨界區\n");
        
        sem_op(semid, 1);  // V 操作
        
        waitpid(pid, NULL, 0);
        
        // 刪除訊號量
        semctl(semid, 0, IPC_RMID);
    }
    
    printf("完成\n");
    return 0;
}
```

### 5.6 插座通訊

Socket 是網路和本地行程間通訊的重要機制。

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <sys/socket.h>
#include <sys/un.h>
#include <sys/wait.h>

#define SOCKET_PATH "/tmp/test_socket"

int main() {
    // 建立 socket
    int server_fd = socket(AF_UNIX, SOCK_STREAM, 0);
    
    if (server_fd == -1) {
        perror("socket");
        exit(EXIT_FAILURE);
    }
    
    // 綁定 socket
    struct sockaddr_un addr;
    memset(&addr, 0, sizeof(addr));
    addr.sun_family = AF_UNIX;
    strncpy(addr.sun_path, SOCKET_PATH, sizeof(addr.sun_path) - 1);
    
    unlink(SOCKET_PATH);  // 移除舊的 socket
    
    if (bind(server_fd, (struct sockaddr *)&addr, sizeof(addr)) == -1) {
        perror("bind");
        close(server_fd);
        exit(EXIT_FAILURE);
    }
    
    // 監聽連線
    if (listen(server_fd, 5) == -1) {
        perror("listen");
        close(server_fd);
        unlink(SOCKET_PATH);
        exit(EXIT_FAILURE);
    }

    }

    // 清理

    close(server_fd);
```

```c
#include <stdio.h>
#include <stdlib.h>
#include <signal.h>
#include <unistd.h>

void handler(int signum) {
    printf("收到訊號：%d\n", signum);
}

int main() {
    // 建立空的訊號集合
    sigset_t set, oldset, pending;
    sigemptyset(&set);
    sigemptyset(&pending);
    
    // 新增要阻塞的訊號
    sigaddset(&set, SIGINT);
    sigaddset(&set, SIGQUIT);
    
    // 設定訊號阻塞
    if (sigprocmask(SIG_BLOCK, &set, &oldset) == -1) {
        perror("sigprocmask");
        return 1;
    }
    
    printf("訊號已被阻塞。10 秒後解除...\n");
    printf("在此期間嘗試發送 SIGINT (Ctrl+C) 或 SIGQUIT (Ctrl+\\)\n");
    
    // 設定處理器
    signal(SIGINT, handler);
    signal(SIGQUIT, handler);
    
    // 等待一段時間
    sleep(10);
    
    // 檢查待處理訊號
    sigpending(&pending);
    
    if (sigismember(&pending, SIGINT)) {
        printf("SIGINT 待處理\n");
    }
    if (sigismember(&pending, SIGQUIT)) {
        printf("SIGQUIT 待處理\n");
    }
    
    // 解除阻塞
    printf("解除訊號阻塞...\n");
    sigprocmask(SIG_SETMASK, &oldset, NULL);
    
    printf("完成\n");
    return 0;
}
```

### 8.3 非阻塞 I/O 與多路復用

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <sys/socket.h>
#include <netinet/in.h>
#include <arpa/inet.h>
#include <sys/select.h>
#include <fcntl.h>

#define PORT 9000
#define MAX_CLIENTS 10

int main() {
    int server_fd, client_fds[MAX_CLIENTS];
    struct sockaddr_in address;
    fd_set read_fds;
    int max_fd, activity;
    
    // 初始化客戶端陣列
    for (int i = 0; i < MAX_CLIENTS; i++) {
        client_fds[i] = -1;
    }
    
    // 建立 socket
    server_fd = socket(AF_INET, SOCK_STREAM, 0);
    
    // 設定為非阻塞
    int flags = fcntl(server_fd, F_GETFL, 0);
    fcntl(server_fd, F_SETFL, flags | O_NONBLOCK);
    
    // 綁定
    int opt = 1;
    setsockopt(server_fd, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));
    
    memset(&address, 0, sizeof(address));
    address.sin_family = AF_INET;
    address.sin_addr.s_addr = INADDR_ANY;
    address.sin_port = htons(PORT);
    
    bind(server_fd, (struct sockaddr *)&address, sizeof(address));
    listen(server_fd, 5);
    
    printf("非阻塞伺服器監聽連接埠 %d...\n", PORT);
    
    while (1) {
        // 清除並設定 fd set
        FD_ZERO(&read_fds);
        FD_SET(server_fd, &read_fds);
        
        max_fd = server_fd;
        
        for (int i = 0; i < MAX_CLIENTS; i++) {
            if (client_fds[i] != -1) {
                FD_SET(client_fds[i], &read_fds);
                if (client_fds[i] > max_fd) {
                    max_fd = client_fds[i];
                }
            }
        }
        
        // 等待活動
        activity = select(max_fd + 1, &read_fds, NULL, NULL, NULL);
        
        if (activity < 0) {
            perror("select");
            break;
        }
        
        // 檢查新連線
        if (FD_ISSET(server_fd, &read_fds)) {
            struct sockaddr_in client_addr;
            socklen_t client_len = sizeof(client_addr);
            int new_fd = accept(server_fd, (struct sockaddr *)&client_addr, &client_len);
            
            if (new_fd >= 0) {
                fcntl(new_fd, F_SETFL, flags | O_NONBLOCK);
                
                // 新增到客戶端陣列
                for (int i = 0; i < MAX_CLIENTS; i++) {
                    if (client_fds[i] == -1) {
                        client_fds[i] = new_fd;
                        printf("新連線：%s:%d (fd=%d)\n",
                               inet_ntoa(client_addr.sin_addr),
                               ntohs(client_addr.sin_port),
                               new_fd);
                        break;
                    }
                }
            }
        }
        
        // 檢查客戶端資料
        for (int i = 0; i < MAX_CLIENTS; i++) {
            if (client_fds[i] != -1 && FD_ISSET(client_fds[i], &read_fds)) {
                char buffer[1024];
                ssize_t n = read(client_fds[i], buffer, sizeof(buffer) - 1);
                
                if (n <= 0) {
                    printf("客戶端 fd=%d 離開\n", client_fds[i]);
                    close(client_fds[i]);
                    client_fds[i] = -1;
                } else {
                    buffer[n] = '\0';
                    printf("fd=%d 收到：%s", client_fds[i], buffer);
                    
                    // 回覆
                    write(client_fds[i], "OK\n", 3);
                }
            }
        }
    }
    
    close(server_fd);
    return 0;
}
```

```c
}

int main() {
    struct sigevent sev;
    struct itimerspec its;
    timer_t timerid;
    
    // 設定計時器
    memset(&sev, 0, sizeof(sev));
    sev.sigev_notify = SIGEV_THREAD;
    sev.sigev_notify_function = timer_handler;
    sev.sigev_notify_attributes = NULL;
    sev.sigev_value.sival_ptr = &timerid;
    
    // 建立計時器
    if (timer_create(CLOCK_REALTIME, &sev, &timerid) == -1) {
        perror("timer_create");
        return 1;
    }
    
    // 設定週期：1 秒後開始，每 500ms 觸發
    its.it_value.tv_sec = 1;
    its.it_value.tv_nsec = 0;
    its.it_interval.tv_sec = 0;
    its.it_interval.tv_nsec = 500000000;  // 500ms
    
    if (timer_settime(timerid, 0, &its, NULL) == -1) {
        perror("timer_settime");
        return 1;
    }
    
    printf("計時器已啟動，等待 5 秒...\n");
    sleep(5);
    
    // 停止計時器
    its.it_value.tv_sec = 0;
    its.it_value.tv_nsec = 0;
    timer_settime(timerid, 0, &its, NULL);
    
    printf("計時器已停止，總共觸發 %d 次\n", timer_count);
    
    timer_delete(timerid);
    return 0;
}
```

### 10.1 錯誤處理機制

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <errno.h>

void handle_error(const char *context) {
    fprintf(stderr, "錯誤：%s\n", context);
    fprintf(stderr, "系統錯誤：%s (errno=%d)\n", strerror(errno), errno);
}

FILE *safe_fopen(const char *path, const char *mode) {
    FILE *fp = fopen(path, mode);
    if (fp == NULL) {
        char msg[256];
        snprintf(msg, sizeof(msg), "無法開啟檔案：%s", path);
        handle_error(msg);
    }
    return fp;
}

void *safe_malloc(size_t size) {
    void *ptr = malloc(size);
    if (ptr == NULL) {
        handle_error("記憶體配置失敗");
        exit(EXIT_FAILURE);
    }
    return ptr;
}

int safe_divide(int a, int b) {
    if (b == 0) {
        errno = EDOM;  // 數學引數超出範圍
        handle_error("除數為零");
        return 0;
    }
    return a / b;
}

int main() {
    // 測試檔案錯誤
    FILE *fp = safe_fopen("/nonexistent/file.txt", "r");
    
    // 測試記憶體錯誤
    void *mem = safe_malloc(1024);
    free(mem);
    
    // 測試除法錯誤
    int result = safe_divide(10, 0);
    printf("10 / 0 = %d\n", result);
    
    // 自定義錯誤
    errno = EACCES;
    handle_error("權限不足");
    
    return 0;
}
```

### 10.2 追蹤與日誌

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <time.h>
#include <syslog.h>
#include <stdarg.h>

typedef enum {
    LOG_DEBUG,
    LOG_INFO,
    LOG_WARNING,
    LOG_ERROR
} log_level_t;

const char *level_strings[] = {
    "DEBUG", "INFO", "WARNING", "ERROR"
};

void log_message(log_level_t level, const char *format, ...) {
    time_t now = time(NULL);
    struct tm *tm_info = localtime(&now);
    char time_str[32];
    strftime(time_str, sizeof(time_str), "%Y-%m-%d %H:%M:%S", tm_info);
    
    fprintf(stdout, "[%s] [%s] ", time_str, level_strings[level]);
    
    va_list args;
    va_start(args, format);
    vfprintf(stdout, format, args);
    va_end(args);
    
    fprintf(stdout, "\n");
    
    // 同步到系統日誌
    if (level >= LOG_ERROR) {
        openlog("myapp", LOG_PID, LOG_USER);
        va_start(args, format);
        vsyslog(LOG_ERR, format, args);
        va_end(args);
        closelog();
    }
}

#define LOG_DEBUG(...) log_message(LOG_DEBUG, __VA_ARGS__)
#define LOG_INFO(...)  log_message(LOG_INFO, __VA_ARGS__)
#define LOG_WARNING(...) log_message(LOG_WARNING, __VA_ARGS__)
#define LOG_ERROR(...) log_message(LOG_ERROR, __VA_ARGS__)

int main() {
    LOG_DEBUG("這是一條除錯訊息");
    LOG_INFO("程式啟動");
    LOG_WARNING("記憶體使用率高：%d%%", 85);
    LOG_ERROR("無法連線到資料庫：%s", "Connection refused");
    
    return 0;
}
```

### 10.3 記憶體錯誤檢測

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define DEBUG_MEMORY

#ifdef DEBUG_MEMORY

static size_t total_allocated = 0;
static size_t total_freed = 0;
static int allocation_count = 0;

void *debug_malloc(size_t size, const char *file, int line) {
    void *ptr = malloc(size);
    if (ptr) {
        total_allocated += size;
        allocation_count++;
        printf("[ALLOC] %s:%d - %zu 位元組 (總計: %zu, 計數: %d)\n", 
               file, line, size, total_allocated, allocation_count);
    }
    return ptr;
}

void *debug_calloc(size_t nmemb, size_t size, const char *file, int line) {
    void *ptr = calloc(nmemb, size);
    if (ptr) {
        total_allocated += nmemb * size;
        allocation_count++;
        printf("[CALLOC] %s:%d - %zu 位元組 (總計: %zu, 計數: %d)\n", 
               file, line, nmemb * size, total_allocated, allocation_count);
    }
    return ptr;
}

void debug_free(void *ptr, size_t size, const char *file, int line) {
    if (ptr) {
        total_freed += size;
        allocation_count--;
        printf("[FREE] %s:%d - %zu 位元組 (釋放: %zu, 計數: %d)\n", 
               file, line, size, total_freed, allocation_count);
    }
    free(ptr);
}

#define malloc(size) debug_malloc(size, __FILE__, __LINE__)
#define calloc(nmemb, size) debug_calloc(nmemb, size, __FILE__, __LINE__)
#define free(ptr, size) debug_free(ptr, size, __FILE__, __LINE__)

#endif

int main() {
    printf("=== 記憶體追蹤測試 ===\n");
    
    int *p1 = (int *)malloc(sizeof(int) * 10);
    int *p2 = (int *)calloc(5, sizeof(int));
    char *p3 = (char *)malloc(100);
    
    printf("\n目前配置：%d 個區塊\n", allocation_count);
    
    free(p1, sizeof(int) * 10);
    free(p2, sizeof(int) * 5);
    free(p3, 100);
    
    printf("\n最終配置：%d 個區塊\n", allocation_count);
    
    return 0;
}
```
